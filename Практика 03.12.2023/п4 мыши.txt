            for (int i = 0; i < mouse.Length; i++) { mouse[i] = 1; }
            while (mouse.Count(j => j == 1) > 1)
            {
                x += m;
                Console.WriteLine(x);
                if (mouse[(x % k)] == 1) { mouse[(x % k)] = 0; }
                else
                {
                    while (mouse[(x % k)] == 0) { x++; }
                    mouse[(x % k)] = 0;
                    x += 1;
                }
            }
            y = Array.IndexOf(mouse, 1);
            Console.WriteLine((k + bel - y) % k);
